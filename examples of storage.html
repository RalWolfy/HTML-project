    <head>

    <script>

        function setItem() {
            if (typeof(Storage) !== "undefined") {
                localStorage.setItem("name", "Ride");
                document.getElementById("name").innerHTML = localStorage.getItem("name");
            } else {
                document.getElementById("name").innerHTML = "Sorry, your browser does not support local storage!";
            }
        }

    </script>

    <script>

        function clickCounter() {
            if (typeof(Storage) !== "undefined") {
                if(localStorage.clickcount) {
                    localStorage.clickcount = Number(localStorage.clickcount)+1;
                    } else {
                        localStorage.clickcount = 1;
                    }
                document.getElementById("result").innerHTML = "You have clicked the button" + localStorage.clickcount + "time(s)"; 
              } else {
                document.getElementById("result").innerHTML = "Sorry, your browser does not support the local storage!";
              }
            }

    </script>

    <script>

        function removeItem() {
            if (typeof(Storage) !== "undefined") {
                localStorage.setItem("name", "Ride");
                document.getElementById("name").innerHTML = localStorage.removeItem("name");
            } else {
                document.getElementById("name").innerHTML = "Sorry, your browser does not support the local storage!";
            }
        }

    </script>
