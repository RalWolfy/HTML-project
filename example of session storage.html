            <script>
        
                function setItem() {
                    if (typeof(Storage) !== "undefined") {
                        sessionStorage.setItem("name", "Ride");
                        document.getElementById("name").innerHTML = sessionStorage.getItem("name");
                    } else {
                        document.getElementById("name").innerHTML = "Sorry, your browser does not support session storage!";
                    }
                }
        
            </script>
        
            <script>
        
                function clickCounter() {
                    if (typeof(Storage) !== "undefined") {
                        if(sessionStorage.clickcount) {
                            sessionStorage.clickcount = Number(sessionStorage.clickcount)+1;
                            } else {
                                sessionStorage.clickcount = 1;
                            }
                        document.getElementById("result").innerHTML = "You have clicked the button" + sessionStorage.clickcount + "time(s)"; 
                      } else {
                        document.getElementById("result").innerHTML = "Sorry, your browser does not support the session storage!";
                      }
                    }
        
            </script>
        
            <script>
        
                function removeItem() {
                    if (typeof(Storage) !== "undefined") {
                        sessionStorage.setItem("name", "Ride");
                        document.getElementById("name").innerHTML = sessionStorage.removeItem("name");
                    } else {
                        document.getElementById("name").innerHTML = "Sorry, your browser does not support the session storage!";
                    }
                }
        
            </script>
